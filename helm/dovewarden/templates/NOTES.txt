1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "dovewarden.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "dovewarden.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "dovewarden.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "{{ include "dovewarden.selectorLabels" . }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check the pod status:
  kubectl --namespace {{ .Release.Namespace }} get pods -l "{{ include "dovewarden.selectorLabels" . }}"

3. View logs:
  kubectl --namespace {{ .Release.Namespace }} logs -l "{{ include "dovewarden.selectorLabels" . }}"

{{- if .Values.metricsService.enabled }}

4. Get the metrics endpoint:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "{{ include "dovewarden.selectorLabels" . }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Metrics available at http://127.0.0.1:9090/metrics"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 9090:9090
{{- end }}

{{- if .Values.serviceMonitor.enabled }}

5. ServiceMonitor created for Prometheus integration:
  kubectl get servicemonitor -n {{ .Release.Namespace }} {{ include "dovewarden.fullname" . }}
{{- end }}

    Please set replicaCount: 1 when using inmemory redis mode.
{{- end }}
